function cetakTandaTerima(nama, nopol, layanan) { let namaPengambil = prompt("Masukkan Nama Pengambil Berkas:", nama); if (namaPengambil == null || namaPengambil.trim() === "") return; let w = window.open('', '_blank'); w.document.write(` Tanda Terima Berkas

AZ-ZAHRA BIRO JASA

Bukti Penyerahan Dokumen Kendaraan

Telah diserahkan kembali berkas asli kepada pelanggan dengan rincian berikut:

Nama Pelanggan: ${nama}

No. Polisi Kendaraan: ${nopol}

Jenis Layanan: ${layanan}

Tanggal Ambil: ${new Date().toLocaleDateString('id-ID', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' })}

Yang Menyerahkan, Yang Menerima,
( AZ-ZAHRA BIRO JASA ) ( ${namaPengambil} )
`); w.document.close(); }
