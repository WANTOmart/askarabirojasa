function bayar(index){ let data = JSON.parse(localStorage.getItem("dataTransaksi")); let tambahan = prompt("Masukkan pembayaran tambahan"); if(tambahan===null) return; tambahan = Number(tambahan); data[index].dp = Number(data[index].dp) + tambahan; data[index].sisa = Number(data[index].biayaJasa) - Number(data[index].dp); data[index].pembayaran = data[index].sisa <= 0 ? "Lunas" : "Belum Lunas"; localStorage.setItem( "dataTransaksi", JSON.stringify(data) ); renderTabel(); } function hapus(index){ if(confirm("Hapus transaksi ini?")){ let data = JSON.parse(localStorage.getItem("dataTransaksi")); data.splice(index,1); localStorage.setItem( "dataTransaksi", JSON.stringify(data) ); renderTabel(); } } function cetakNota(index){ let data = JSON.parse(localStorage.getItem("dataTransaksi")); let t = data[index]; let w = window.open("","","width=700,height=700"); w.document.write(` Nota

NOTA TRANSAKSI

No Nota : ${t.nota}

Tanggal : ${t.tanggal}

Pelanggan : ${t.pelanggan}

No Polisi : ${t.noPolisi}

Kendaraan : ${t.kendaraan}

Layanan : ${t.layanan}

Total Biaya : Rp ${Number(t.biayaJasa).toLocaleString('id-ID')}

DP : Rp ${Number(t.dp).toLocaleString('id-ID')}

Sisa : Rp ${Number(t.sisa).toLocaleString('id-ID')}

${t.pembayaran}

`); w.print(); }
